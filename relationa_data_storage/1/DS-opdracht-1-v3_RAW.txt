  naam varchar(255) [not null]
  adres_id int [not null]
  postbus_id int [not null]
  email varchar(255) [not null]
  mobiel_nummer int [not null]
  vast_nummer int [not null]
  contactpersonen_id int [not null]

  // Mogelijke eigenschappen (all optional)
  
  // Type C
  overige_adressen text
  aanbiedingen text
  aangesloten_bij text
  algemene_bedrijfsinfo text
  brochure text
  certificaten_keurmerk text
  dealer_van text
  dealers text
  dochterbedrijf_van text
  downloads text
  folders text
  huismerken text
  importeur_van text

  app varchar(255)

  advies boolean

  directe_link_met_bestellijsten boolean
  eco_producten boolean

  garantie boolean
  gastenboek boolean
  gerechten_recepten boolean
  glutenvrij boolean
  halal boolean
  handleidingen boolean
  installatie_montage boolean
  kerstpakketten boolean

  created_at timestamp [default: `CURRENT_TIMESTAMP`]
  updated_at timestamp [default: `CURRENT_TIMESTAMP`, note: 'ON UPDATE CURRENT_TIMESTAMP']
}

Table adres {
  id int [primary key, increment]
  straatnaam varchar(255) [not null]
  nummer int [not null] 
  postcode_id int [not null]
  woonplaats varchar(255) [not null]
  gemeente_id int [not null] 
  created_at timestamp [default: `CURRENT_TIMESTAMP`]
  updated_at timestamp [default: `CURRENT_TIMESTAMP`, note: 'ON UPDATE CURRENT_TIMESTAMP']
}

Table postbus {
  id int [primary key, increment]
  postbusnummer int [not null]
  postcode_id int [not null]
  created_at timestamp [default: `CURRENT_TIMESTAMP`]
  updated_at timestamp [default: `CURRENT_TIMESTAMP`, note: 'ON UPDATE CURRENT_TIMESTAMP']
}

Table gemeente {
  id int [primary key, increment]
  naam varchar(255) [not null]
  provincie_id int [not null]
}

Table provincie {
  id int [primary key, increment]
  naam varchar(255) [not null]
  land_id int [not null]
}

Table land {
  id int [primary key, increment]
  naam varchar(255) [not null]
}

Table postcode {
  id int [primary key, increment]
  postcode varchar(6) [not null]
  straatnaam varchar(255) [not null]
  woonplaats varchar(255) [not null]
  coordinates point [not null]
  Note: '''
    Spatial data table using MySQL POINT type
    - SRID 4326 (WGS 84 coordinate system)
    - SPATIAL INDEX on coordinates
    - Format: POINT(longitude latitude)
  '''
}

Table contactpersonen {
  id int [primary key, increment]
  contactpersoon_id int [not null]
}

Table contactpersoon {
  id int [primary key, increment]
  naam varchar(255) [not null]
  email varchar(255) [not null]
  mobiel_nummer int [not null]
  werkzaamheden_id int [not null]
}

Table werkzaamheden {
  id int [primary key, increment]
  afdeling_id int [not null]
  taak_id int [not null]
}

Table afdeling {
  id int [primary key, increment]
  naam varchar(255) [not null]
}

Table taak {
  id int [primary key, increment]
  naam varchar(255) [not null]
}
